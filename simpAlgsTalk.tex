\frame{\frametitle{Adams sseq for simplicial algebras} 
\vspace{-1cm}{\huge \[H_Q^*:s\calC\to\calP\scrL^1\]}

$\calP\scrL^1$ is category with objects $W^*$:
\begin{itemise}
\setlength{\parindent}{.25in}
\item $[\DASH,\DASH]:W^n\otimes W^m \to W^{n+m+1}$ \pause
\begin{enumerate}\squishlist
\setlength{\parindent}{.25in}
\item Jacobi identity
\item (anti)-symmetric, (\emph{not} ``alternating'') \pause
\end{enumerate}
\item linear operations $P^i:W^n\to W^{n+i+1}$ \pause
\begin{enumerate}\squishlist
\setlength{\parindent}{.25in}
\item zero unless $2\leq i\leq n$
\item $P^{|x|}x=[x,x]$
\item an Adem relation \pause
\end{enumerate}
\item ``Cartan relation'': $[x,P^iy]=0$
\end{itemise}

}
