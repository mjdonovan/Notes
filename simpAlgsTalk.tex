\frame{\frametitle{Adams sseq for simplicial algebras} 

For $X\in s\calC$:
\[E_2^{st}=\Ext^s_{\scrK^1}(H^*_QX,H^*_QS^t)\implies \pi_{t-s}(X{\hat{\phantom{e}}})\]

$\scrK^1$ is category with objects $W^*$:
\begin{itemise}
\setlength{\parindent}{.25in}
\item $[\DASH,\DASH]:W^n\otimes W^m \to W^{n+m+1}$
\begin{enumerate}\squishlist
\setlength{\parindent}{.25in}
\item Jacobi identity
\item (anti)-symmetric, \emph{not alternating}
\end{enumerate}
\item linear operations $P^i:W^n\to W^{n+i+1}$
\begin{enumerate}\squishlist
\setlength{\parindent}{.25in}
\item zero unless $2\leq i\leq n$
\item $P^{|x|}x=[x,x]$
\item an Adem relation
\end{enumerate}
\item ``Cartan relation'': $[x,P^iy]=0$
\end{itemise}

}
