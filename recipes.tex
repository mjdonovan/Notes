\recipe{White bread with poolish}
{\item\ingstep{
\ingred{250g}{all-purpose flour}
\ingred{.2g}{yeast  (a pinch)}
\ingred{250g}{80\degree\ water}
}%
{mix the poolish}
\item Ferment poolish 12-14 hours. Should be bubbly \& tripled in volume.
\item
\ingstep{
\ingred{250g}{all-purpose flour}
\ingred{1.5g}{yeast}
\ingred{1/4 cup}{10 grain mix}}%
{mix dry ingredients}
\item 
\ingstep{
\ingred{125g}{105\degree\ water}}%
{water}
\item Pour dry ingredients over poolish, then add the water. Mix (cut and fold).
\item Fold 2-3 times in first hour of a 2-3 hour bulk fermentation --- dough should rise $\times2.5$.
\item Shape, proof for 1 hour --- preheat oven to 475\degree, 15 minutes after shaping.
\item Bake 30+10 minutes at 475{\ensuremath{^\circ}}.}

\recipe{Pizza dough ($\times4$)}%
{
\item\ingstep{
\ingred{450g}{white flour}
\ingred{310g}{95\degree\ water}
}%
{incorporate in mixing bowl}
\item Autolyse 20 minutes.
\item\ingstep{
\ingred{10g}{salt}
\ingred{100g}{80\% starter}
}%
{cut and fold}
\item Fold after 30 minutes.
\item At 60 minutes, lightly oil dough bucket, fold into bucket, and turn in oil.
\item Bulk ferment (approx 12 hours, should rise $\times$2-2.5).
\item Shape into 4 balls.
\item $\begin{cases}
\textup{Refrigerate 6 hours, or}&\\
\textup{leave to rise for 1 hour, then refrigerate for 30 minutes.}&
%\\,&\textup{if }
\end{cases}$
}

\recipe{Pita ($\times8$)}%
{
\item\ingstep{
\ingred{260g}{all-purpose flour}%was 200
\ingred{293g}{white whole wheat}%was 225g
\ingred{13g}{castor sugar}%was 10g
\ingred{10g}{salt}%was 8g
}%
{dry ingredients}
\item\ingstep{
\ingred{318g}{90\degree\ water}%was 245g
\ingred{180g}{80\% starter}
\ingred{26g}{olive oil}%was 20g
}%
{wet ingredients}
\item Cut and fold, then knead for 6 minutes.
\item Turn in oiled bowl, rest 30 minutes.
\item Form into 6-8 balls.
\item Refrigerate 8-16 hours.
\item \textbf{Optional:} Freeze. Defrost later for 3 hours, or 8 hours in fridge.
\item Bake at 450\degree.
}
