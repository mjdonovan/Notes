\recipe{Overnight country blonde}
{\item
\ingstep{
\ingred{404g}{all-purpose flour}
\ingred{36g}{whole wheat/rye blend}
\ingred{1Tbsp}{gluten}
\ingred{1Tbsp}{flaxseed meal}
\ingred{1/4 cup}{10 grain mix}}%
{dry ingredients}

\item 
\ingstep{
\ingred{342g}{95{\ensuremath{^\circ}} water}}%
{water}
\item Autolyse 15-20 mins.
\item 
\ingstep{
\ingred{10g}{salt}
\ingred{140g}{80\% starter}}%
{salt and starter}
\item Fold 3-4 times.
\item Bulk ferment (approx 12 hours, should triple).
\item Shape, rest for 3 hours (a little less?).
\item Bake 30+10 minutes at 475{\ensuremath{^\circ}}.}

\recipe{Pizza dough}%
{
\item\ingstep{
\ingred{450g}{white flour}
\ingred{310g}{95\degree\ water}
}%
{incorporate in mixing bowl}
\item Autolyse 20 minutes.
\item\ingstep{
\ingred{10g}{salt}
\ingred{100g}{80\% levain}
}%
{cut and fold}
\item Fold after 30 minutes.
\item At 60 minutes, lightly oil dough bucket, fold into bucket, and turn in oil.
\item Bulk ferment (approx 12 hours, should scale by 2-2.5).
\item Shape into 4 balls, refrigerate 6 hours.
\item \textbf{Alternative schedule} for use before 6 hours: leave to rise for 1 hour, then refrigerate for 30 minutes.
}

\recipe{Pita}%
{
\item\ingstep{
\ingred{200g}{all-purpose flour}
\ingred{225g}{white whole wheat}
\ingred{10g}{castor sugar}
\ingred{8g}{salt}
}%
{dry ingredients}
\item\ingstep{
\ingred{245g}{90\degree\ water}
\ingred{180g}{80\% starter}
\ingred{20g}{olive oil}
}%
{wet ingredients}
\item Cut and fold, then knead for 6 minutes.
\item Turn in oiled bowl, rest 30 minutes.
\item Form into 6-8 balls.
\item Refridgerate 6-8 hours.
\item \textbf{Optional:} Freeze. Defrost for 4 hours, or 8 hours in fridge.
\item Bake at 450\degree.
}
